import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib import font_manager, rc
from matplotlib.dates import DateFormatter
from sqlalchemy import create_engine

#    (Windows:  )
font_path = "C:/Windows/Fonts/malgun.ttf"
font = font_manager.FontProperties(fname=font_path).get_name()
rc('font', family=font)
plt.rcParams['axes.unicode_minus'] = False  #   

# DB  
user = 'root'
password = " "
host = 'localhost'
port = 3306
db_name = 'de_1_weather'
db_url = f'mysql+pymysql://{user}:{password}@{host}:{port}/{db_name}'
engine = create_engine(db_url)

#  
query = "SELECT , ` `, ` `, `  `, `  ` FROM weather_7 ORDER BY "
df = pd.read_sql(query, con=engine)

#   datetime  
df[''] = pd.to_datetime(df[''], errors='coerce', format='%Y.%m.%d')

# 
fig, axes = plt.subplots(1, 2, figsize=(15, 6))

# /  
df_temp_melted = pd.melt(df, id_vars=[''], value_vars=[' ', ' '],
                         var_name=' ', value_name='')

sns.lineplot(data=df_temp_melted, x='', y='', hue=' ',
             marker='o', palette='Set2', ax=axes[0])
axes[0].set_title(' / ', fontsize=14)
axes[0].set_xlabel('')
axes[0].set_ylabel(' (C)')
axes[0].tick_params(axis='x', rotation=45)
axes[0].grid(True)

#    
df_rain_melted = pd.melt(df, id_vars='', value_vars=['  ', '  '],
                         var_name='', value_name=' ')

sns.barplot(data=df_rain_melted, x='', y=' ', hue='',
            palette=['#EF9A9A', '#90CAF9'], errorbar=None, ax=axes[1])

axes[1].xaxis.set_major_formatter(DateFormatter('%Y-%m-%d'))
axes[1].set_title('   (/)', fontsize=14)
axes[1].set_xlabel('')
axes[1].set_ylabel('  (%)')
axes[1].tick_params(axis='x', rotation=45)
axes[1].grid(True)
axes[1].legend()

plt.tight_layout()
plt.show()

#  
df_rain = df[['', '  ', '  ']].set_index('')
plt.figure(figsize=(10, 6))
sns.heatmap(df_rain.T, cmap='YlGnBu', annot=True, fmt='d',
            cbar=True, linewidths=0.5)

plt.title('   (, )', fontsize=14)
plt.xlabel('')
plt.ylabel('')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
